bool main_dfs(std::vector<Vertex> &vertices, std::vector<bool> &seen, int start)
{
    std::cout << start + 1 << " ";
    seen[start] = true;
    for (int i = 0; i < (int)vertices[start].adjacentVertex.size(); i++)
    {
        int v = vertices[start].adjacentVertex[i];
        if (!seen[v])
        {
            main_dfs(vertices, seen, v);
        }
    }
    return true;
}
